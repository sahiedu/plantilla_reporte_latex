\subsection{Modelo dinámico de un robot manipulador de 2 GDL}

El modelo dinámico de un robot manipulador de $n$ grados de libertad está dado por la siguiente expresión:

\begin{equation}
	M(\mathbf{q})\ddot{\mathbf{q}} + C(\mathbf{q}, \dot{\mathbf{q}})\dot{\mathbf{q}} + \mathbf{g}(\mathbf{q}) = \mathbf{\tau}
\end{equation}

Específicamente, para un robot manipulador de dos grados de libertad se tiene:

\begin{equation}
	\begin{bmatrix}
		M_{11}(\mathbf{q}) & M_{12}(\mathbf{q})\\
		M_{21}(\mathbf{q}) & M_{22}(\mathbf{q})\\
	\end{bmatrix}\ddot{\mathbf{q}} 
	+
	\begin{bmatrix}
		C_{11}(\mathbf{q}, \dot{\mathbf{q}}) & C_{12}(\mathbf{q}, \dot{\mathbf{q}})\\
		C_{21}(\mathbf{q}, \dot{\mathbf{q}}) & C_{22}(\mathbf{q}, \dot{\mathbf{q}})\\
	\end{bmatrix}\dot{\mathbf{q}}
	+
	\begin{bmatrix}
		g_{1}(\mathbf{q})\\
		g_{2}(\mathbf{q})\\
	\end{bmatrix}
	=
	\mathbf{\tau},
\end{equation}

Donde:

\begin{equation}
	\begin{aligned}
		M_{11}(\mathbf{q}) &= m_{1}l_{c1}^{2} + m_{2}\left[l_{1}^{2} + l_{c2}^{2} + 2l_{1}l_{c2}\cos(q_{2}) \right]  + I_{1} + I_{2}\\
		M_{12}(\mathbf{q}) &= m_{2}\left[l_{c2}^{2} + l_{1}l_{c2}\cos(q_{2})\right] + I_{2}\\
		M_{21}(\mathbf{q}) &= m_{2}\left[l_{c2}^{2} + l_{1}l_{c2}\cos(q_{2})\right] + I_{2}\\
		M_{22}(\mathbf{q}) &= m_{2}l_{c2}^{2} + I_{2}\\
		\\
		C_{11}(\mathbf{q}, \dot{\mathbf{q}}) &= -m_{2}l_{1}l_{c2}\sen(q_{2})\dot{q}_{2}\\
		C_{12}(\mathbf{q}, \dot{\mathbf{q}}) &= -m_{2}l_{1}l_{c2}\sen(q_{2})\left[\dot{q}_{1} +\dot{q}_{2}\right] \\
		C_{21}(\mathbf{q}, \dot{\mathbf{q}}) &= m_{2}l_{1}l_{c2}\sen(q_{2})\dot{q}_{1}\\
		C_{22}(\mathbf{q}, \dot{\mathbf{q}}) &= 0\\
		\\
		g_{1}(\mathbf{q}) &= \left[m_{1}l_{c1} + m_{2}l_{1}\right]g\sen(q_{1}) + m_{2}l_{c2}g\sen(q_{1}+q_{2}) \\
		g_{2}(\mathbf{q}) &= m_{2}l_{c2}g\sen(q_{1}+q_{2}).\\
	\end{aligned}
\end{equation}

Considerando a $\mathbf{q}=[q_{1}, q_{2}]^{\top}$ y $\dot{\mathbf{q}}=[\dot{q}_{1}, \dot{q}_{2}]^{\top}$ como variables de estado, el modelo dinámico del robot se puede reescribir de la siguiente forma:

\begin{equation}
	\frac{d}{dt}
	\begin{bmatrix}
		\mathbf{q}\\
		\\
		\dot{\mathbf{q}}\\
	\end{bmatrix}
	=
	\begin{bmatrix}
		\dot{\mathbf{q}}\\
		\\
		M(\mathbf{q})^{-1}\left[\mathbf{\tau} - C(\mathbf{q}, \dot{\mathbf{q}})\dot{\mathbf{q}} - \mathbf{g}(\mathbf{q})\right] 
	\end{bmatrix}
\end{equation}

Este es un ejemplo de teoremas.

\subsubsection{Varias ecuaciones}\index{teoremas!varias ecuaciones}
Este es un teorema que consta de varias ecuaciones.

\begin{theorem}[Nombre del teorema]
	En $E=\mathbb{R}^n$ todas las normas son equivalentes. Se tienen las siguientes propiedades:
	\begin{align}
		& \abs{\norm{\mathbf{x}} - \norm{\mathbf{y}}} \leq \norm{\mathbf{x}- \mathbf{y}}\\
		&  \norm{\sum_{i=1}^n\mathbf{x}_i}\leq \sum_{i=1}^n\norm{\mathbf{x}_i}\quad\text{donde $n$ es un entero finito}
	\end{align}
\end{theorem}

\subsubsection{Una sola línea}\index{teoremas!una sola línea}
Este es un teorema que consta de una sola línea.

\begin{theorem}
	Un conjunto $\mathcal{D}(G)$ es denso en $L^2(G)$, $|\cdot|_0$.
\end{theorem}

%------------------------------------------------

\subsection{Definiciones}\index{definiciones}

Este es un ejemplo de una definición. Una definición puede ser matemática o puede describir un concepto.

\begin{definition}[Nombre de la definición]
	Dado un espacio vectorial $E$, una norma sobre $E$ es una aplicación, denotada por $||\cdot||$, de $E$ en $\mathbb{R}^+=[0,+\infty[$ tal que:
	\begin{align}
		& \norm{\mathbf{x}}=0\ \Rightarrow\ \mathbf{x}=\mathbf{0}\\
		& \norm{\lambda \mathbf{x}}=\abs{\lambda}\cdot \norm{\mathbf{x}}\\
		& \norm{\mathbf{x}+\mathbf{y}}\leq \norm{\mathbf{x}} + \norm{\mathbf{y}}
	\end{align}
\end{definition}
%------------------------------------------------

\subsection{Notación}\index{notación}

\begin{notation}
	Dado un subconjunto abierto $G$ de $\mathbb{R}^n$, el conjunto de funciones $\varphi$ que:
	\begin{enumerate}
		\item Tienen soporte compacto en $G$;
		\item Son infinitamente diferenciables;
	\end{enumerate}
	constituye un espacio vectorial que se denota por $\mathcal{D}(G)$.
\end{notation}
